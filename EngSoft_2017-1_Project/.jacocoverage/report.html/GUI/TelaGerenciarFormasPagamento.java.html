    private void jButton4ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton4ActionPerformed
        try {
            String nome = jTextField4.getText();
            String descricao = jTextField3.getText();
            if(nome.isEmpty() || descricao.isEmpty()){
                JOptionPane.showMessageDialog(null, "Um nome e uma descrição são origatórios para uma forma de pagamento.", "Validação", JOptionPane.ERROR_MESSAGE);
                return;
            }
            FormaPagamento forma = new FormaPagamento();
            forma.setNome(nome);
            forma.setDescricao(descricao);
            Main.getFuncionario().cadastrarFormaPagamento(forma);
            this.carregarFormasPagamento();
        } catch (Exception ex) {
             JOptionPane.showMessageDialog(null, ex.getMessage(), "Erro ao tentar incluir forma de pagamento.", JOptionPane.ERROR_MESSAGE);
               
            Logger.getLogger(TelaGerenciarFormasPagamento.class.getName()).log(Level.SEVERE, null, ex);
        }
    }//GEN-LAST:event_jButton4ActionPerformed

    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton3ActionPerformed
        if(formaPagSelecionada==null) return;
        try {
            String nome = jTextField1.getText();
            String descricao = jTextField2.getText();
            if(nome.isEmpty() || descricao.isEmpty()){
                JOptionPane.showMessageDialog(null, "Um nome e uma descrição são origatórios para uma forma de pagamento.", "Validação", JOptionPane.ERROR_MESSAGE);
                return;
            }
            
            formaPagSelecionada.setNome(nome);
            formaPagSelecionada.setDescricao(descricao);
            Main.getFuncionario().alterarFormaPagamento(formaPagSelecionada);
            this.carregarFormasPagamento();
        } catch (Exception ex) {
             JOptionPane.showMessageDialog(null, ex.getMessage(), "Erro ao tentar incluir forma de pagamento.", JOptionPane.ERROR_MESSAGE);
               
            Logger.getLogger(TelaGerenciarFormasPagamento.class.getName()).log(Level.SEVERE, null, ex);
        }
    }//GEN-LAST:event_jButton3ActionPerformed

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton2ActionPerformed
       if(formaPagSelecionada==null) return;
        try {
            jTextField1.setText("");
            jTextField2.setText("");
            formaPagSelecionada.setApagado(true);

    private void carregarFormasPagamento() {
        Query query = Persistencia.getManager().createQuery("select t from FormaPagamento as t where t.apagado = ? order by t.id desc");
        query.setParameter(1, false);
        DefaultTableModel model = (DefaultTableModel) jTable1.getModel();
        model.setRowCount(0);
        try {
            List<Object> formas = Persistencia.buscar(query);
            for(int i=0; i< formas.size();i++){
                    FormaPagamento forma = (FormaPagamento)formas.get(i);
                    Object[] linha = new Object[]{
                                forma,
                                forma.getNome(),
                                forma.getDescricao()
                            };
                   model.addRow(linha);
            }
            if(!formas.isEmpty()){
                this.formaPagSelecionada = (FormaPagamento)model.getValueAt(0, 0);
                jTable1.setRowSelectionInterval(0, 0);
            }
            
        } catch (Exception ex) {
            JOptionPane.showMessageDialog(null, ex.toString(), "Erro ao obter formas de pagamento", JOptionPane.ERROR_MESSAGE);
            Logger.getLogger(Main.class.getName()).log(Level.SEVERE, null, ex);
        }
    }

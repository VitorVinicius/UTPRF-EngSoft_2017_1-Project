    private void jpDiaAdmissaoStateChanged(javax.swing.event.ChangeEvent evt) {//GEN-FIRST:event_jpDiaAdmissaoStateChanged
        // TODO add your handling code here:
        if(evt.getSource() == this.jpDiaAdmissao){
            int valor = (int) this.jpDiaAdmissao.getValue();
            this.jpDiaAdmissao.setValue(valor<1?1:valor>31?31:valor);
        }
        else
        if(evt.getSource() == this.jpMesAdmissao){
            int valor = (int) this.jpMesAdmissao.getValue();
            this.jpMesAdmissao.setValue(valor<1?1:valor>12?12:valor);
        }
        else
        if(evt.getSource() == this.jpAnoAdmissao){
            int valor = (int) this.jpAnoAdmissao.getValue();
            this.jpAnoAdmissao.setValue(valor<1900?1900:valor>2500?2500:valor);
        }
    }//GEN-LAST:event_jpDiaAdmissaoStateChanged

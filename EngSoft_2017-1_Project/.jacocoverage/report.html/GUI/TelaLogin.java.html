    private void logon(){
        long id = Long.parseLong(jTextField1.getText());
        String senha = String.valueOf( this.jPasswordField1.getPassword());
        Query query = Persistencia.getManager().createQuery("select t from Funcionario as t where t.id = ? and t.senha = ? order by t.id desc");
        query.setParameter(1, id);
        query.setParameter(2, senha);
        query.setMaxResults(1);
        List<Funcionario> funcs = query.getResultList();
        if(!funcs.isEmpty()){
            Main.setFuncionario(funcs.get(0));
            this.dispose();
            new TelaPrincipal().setVisible(true);
        }
        else{
            JOptionPane.showMessageDialog(null,"A combinação id/senha não foi encontrada.", "Logon", JOptionPane.ERROR_MESSAGE);
        }
    }

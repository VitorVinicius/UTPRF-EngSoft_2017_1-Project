    public List<Historico> getHistoricos() throws HistoricoNaoObtidoException {
        if(historicos == null || historicos.isEmpty()){
            try {
                historicos = Entidades.obterHistorico(this);
            } catch (SQLException ex) {
                throw new Excecoes.HistoricoNaoObtidoException("Não foi possível obter o histórico do cliente "+ this.id,ex);
            }
        }
        return historicos;
    }

    public void incluirHistorico(Historico historico) throws OperacaoInvalidaException, SQLException {
        if(historico.getIdPessoa()!=this.id){
           throw new Excecoes.OperacaoInvalidaException("O historico " + historico.getId() + " não pertence ao cliente "+ this.id);
        }
        PreparedStatement  pstmt = Conector.getConexao().prepareStatement("INSERT INTO `locadora`.`historico`\n" +
"(\n" +
"`idPessoa`,\n" +
"`idEquipamento`,\n" +
"`tipoOcorrencia`,\n" +
"`descricao`)\n" +
"VALUES\n" +
"(?,?,?,?);"); 
        pstmt.setLong(1, historico.getIdPessoa());
        pstmt.setLong(2, historico.getIdEquipamento());
        pstmt.setInt(3, historico.getTipoOcorrencia().ordinal());
        pstmt.setString(4, historico.getDescricao());
        pstmt.executeUpdate();
    }
